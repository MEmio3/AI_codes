Java: Reverse a Singly Linked List (Iterative)

Node reverseInPlace(Node head) {
    Node prev = null;                 // 1
    Node curr = head;                 // 2
    while (curr != null) {            // 3
        Node next = curr.next;        // 4
        curr.next = prev;             // 5
        prev = curr;                  // 6
        curr = next;                  // 7
    }
    return prev;                      // 8
}
